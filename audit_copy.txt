PROCEDURE clearaudit AS
    l_date date;
    l_f date;
    l_curdate DATE:=SYSDATE;
    l_step pls_integer;
    l_description VARCHAR2 (200);
BEGIN
    l_step:=1;
    l_description:='AUD: ClearAudit started';
    sys.dbms_system.ksdwrt(2,l_description);

    INSERT INTO LogClearAudit (CurDate, Step, BeginDate, EndDate, Description)
        VALUES (l_curdate, l_step, l_curdate, l_curdate, l_description);
    COMMIT;
    --
    l_step:=l_step+1;
    l_description:='SELECT MIN(nTimeStamp#)INTO l_date FROM Sys.Aud$;';
    INSERT INTO LogClearAudit (CurDate, Step, BeginDate, Description)
        VALUES (l_curdate, l_step, l_curdate, l_description);
    COMMIT;

    SELECT MIN(nTimeStamp#) INTO l_date FROM Sys.Aud$;

    UPDATE /*+ index_desc(LogClearAudit,LogClearAudit_i) */
        LogClearAudit SET EndDate=SYSDATE WHERE Curdate=l_curdate and Step=l_step;
    COMMIT;
--1 week for audit
    l_f:=l_curdate-8;

    WHILE l_date<l_f AND TO_CHAR(SYSDATE,'HH24') <= '07' LOOP
        l_step:=l_step+1;
        l_description:='DELETE FROM Sys.Aud$ WHERE nTimeStamp#<'''||TO_CHAR(l_date,'DD.MM.YYYY HH24:MI:SS')||''';';
        INSERT INTO LogClearAudit (CurDate, Step, BeginDate, Description)
            VALUES (l_curdate, l_step, SYSDATE, l_description);
        COMMIT;

        DELETE FROM Sys.Aud$ WHERE nTimeStamp#<l_date;

    UPDATE /*+ index_desc(LogClearAudit,LogClearAudit_i) */
        LogClearAudit SET EndDate=SYSDATE WHERE Curdate=l_curdate and Step=l_step;
    COMMIT;

        l_date:=l_date+2;
    END LOOP;
    sys.dbms_system.ksdwrt(2,'AUD: ClearAudit succeded');
exception
    WHEN OTHERS THEN sys.dbms_system.ksdwrt(2,'AUD: ClearAudit failed '||sqlerrm(sqlcode));
END;

begin
select sysdate into curdate from dual;
logger(curdate,sysdate, 0,'ia_audit_table_copy procedure started.');
----------------------- BEGIN of Adding Partitions-------------------
select partition_name  BULK COLLECT INTO part_name_tab
   from dba_tab_partitions
   where table_owner='WM_DBA' and table_name = 'AUD$COPY' and upper(partition_name) not like '%MAX%'
   order by partition_position;
if not part_name_tab.exists(1) then
 logger(curdate,sysdate, 20000,'ia_audit_table_copy procedure failed. No partitions found.');
 return;
end if;

part_max_date := to_date(substr(part_name_tab(part_name_tab.last), 12, 8), 'ddmmyyyy');
part_v_diff := part_max_date - trunc(curdate);
if part_v_diff<0 then part_v_diff:=part_cache;part_max_date:=curdate;
    elsif part_v_diff>=0 and part_v_diff<part_cache then part_v_diff:=part_cache-part_v_diff;
    elsif part_v_diff>=part_cache then part_v_diff:=0;
end if;
for i in 1..part_v_diff loop
  part_max_date := part_max_date + 1;
  l_text        := 'alter table aud$copy split partition aud$copy_max at (to_date(''' ||to_char(part_max_date+1,'dd-mm-yyyy')|| ''',''dd-mm-yyyy'')) into
(partition aud$copy_p_' || to_char(part_max_date, 'ddmmyyyy') ||', partition aud$copy_max)';
execute immediate l_text;
logger(curdate,sysdate, 0,'Partition aud$copy_p_' || to_char(part_max_date, 'ddmmyyyy')||' successfully added.');
end loop;

part_old_date:=add_months(trunc(curdate),-part_history);
for i in 1..part_name_tab.last loop
    if to_date(substr(part_name_tab(i),12,8),'ddmmyyyy')<=part_old_date then
    l_text:='alter table aud$copy drop partition '||part_name_tab(i);
    execute immediate l_text;
    logger(curdate,sysdate, 0,'Partition ' || part_name_tab(i)||' successfully dropped.');
    else exit;
    end if;
end loop;
----------------------- END of Adding Partitions -------------------

--------------------------------------------------------------------
--                     Copying rows from AUD$                     --
--------------------------------------------------------------------
logger(curdate,sysdate, 0,'Starting audit loading.');
open c2;
   loop
     fetch c2 into act_row;
     exit when c2%notfound;
     aud_act(act_row.action):=act_row.name;
   end loop;
close c2;
open c1;
  loop
   fetch c1 into aud_row;
   exit when c1%notfound;
    k3:=c1%rowcount;
    k2:=k2+1;
    sys_aud(k2).sessionid:=aud_row.sessionid;
    sys_aud(k2).entryid:=aud_row.entryid;
    sys_aud(k2).statement:=aud_row.statement;
    sys_aud(k2).timestamp#:=aud_row.timestamp#;
    sys_aud(k2).userid:=aud_row.userid;
    sys_aud(k2).userhost:=aud_row.userhost;
    sys_aud(k2).terminal:=aud_row.terminal;
    sys_aud(k2).action#:=aud_row.action#;
    sys_aud(k2).returncode:=aud_row.returncode;
    sys_aud(k2).obj$creator:=aud_row.obj$creator;
    sys_aud(k2).obj$name:=aud_row.obj$name;
    sys_aud(k2).auth$privileges:=aud_row.auth$privileges;
    sys_aud(k2).auth$grantee:=aud_row.auth$grantee;
    sys_aud(k2).new$owner:=aud_row.new$owner;
    sys_aud(k2).new$name:=aud_row.new$name;
    sys_aud(k2).ses$actions:=aud_row.ses$actions;
    sys_aud(k2).ses$tid:=aud_row.ses$tid;
    sys_aud(k2).logoff$lread:=aud_row.logoff$lread;
    sys_aud(k2).logoff$pread:=aud_row.logoff$pread;
    sys_aud(k2).logoff$lwrite:=aud_row.logoff$lwrite;
    sys_aud(k2).logoff$dead:=aud_row.logoff$dead;
    sys_aud(k2).logoff$time:=aud_row.logoff$time;
    sys_aud(k2).comment$text:=aud_row.comment$text;
    sys_aud(k2).clientid:=aud_row.clientid;
    sys_aud(k2).spare1:=aud_row.spare1;
    sys_aud(k2).spare2:=aud_row.spare2;
    sys_aud(k2).obj$label:=aud_row.obj$label;
    sys_aud(k2).ses$label:=aud_row.ses$label;
    sys_aud(k2).priv$used:=aud_row.priv$used;
    sys_aud(k2).sessioncpu:=aud_row.sessioncpu;
    sys_aud(k2).NTIMESTAMP#:=aud_row.NTIMESTAMP#;
    sys_aud(k2).PROXY$SID:=aud_row.PROXY$SID;
    sys_aud(k2).USER$GUID:=aud_row.USER$GUID;
    sys_aud(k2).INSTANCE#:=aud_row.INSTANCE#;
    sys_aud(k2).PROCESS#:=aud_row.PROCESS#;
    sys_aud(k2).XID:=aud_row.XID;
    sys_aud(k2).AUDITID:=aud_row.AUDITID;
    sys_aud(k2).SCN:=aud_row.SCN;
    sys_aud(k2).DBID:=aud_row.DBID;
    sys_aud(k2).sqltext:=aud_row.sqltext;
    if aud_row.action#=192 and not aud_act.exists(192) then
         sys_aud(k2).action_name:='alter synonym';
    else
         sys_aud(k2).action_name:=aud_act(aud_row.action#);
    end if;
    if k2>=v_const then
       forall i in sys_aud.first..sys_aud.last
       insert into aud$copy values sys_aud(i);
       commit;
       sys_aud:=sys_aud_empty;
       k2:=0;
    end if;
   end loop;
close c1;
if sys_aud.first is not null then
    forall i in sys_aud.first .. sys_aud.last
       insert into aud$copy values sys_aud(i);
    commit;
end if;

logger(curdate,sysdate, 0,'ia_audit_table_copy completed. '||k3||' rows inserted.');

exception
    WHEN OTHERS THEN
        logger(curdate,sysdate, sqlcode,sqlerrm(sqlcode));
END;
